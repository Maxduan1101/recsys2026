def controlled_fusion(state, legacy, source_lists, params):
    # state: session-turn context
    # legacy: BM25 ranked list
    # source_lists: dict[source_name] -> ranked list
    # params: calibrated thresholds, gates, quotas, penalties

    intent = detect_intent(state)
    hard_constraints = build_constraint_checker(state)

    # 1. Active source selection
    active_sources = ["legacy"]
    for s, ranking in source_lists.items():
        u_s = predict_source_utility(s, state, intent)
        u_l = predict_source_utility("legacy", state, intent)

        if u_s - u_l > params.delta_source[intent][s]:
            if source_is_reliable(s, intent, params):
                active_sources.append(s)

    active_sources = cap_sources(active_sources, max_nonlegacy=2)

    # 2. Candidate pool
    pool = set(legacy[:params.legacy_pool_size])
    for s in active_sources:
        if s == "legacy":
            continue
        pool.update(source_lists[s][:params.source_pool_size[intent][s]])

    # 3. Remove hard violations
    pool = [d for d in pool if hard_constraints(d)]

    # 4. Calibrated pRel
    p = {}
    for d in pool:
        features = build_candidate_features(
            d, state, legacy, source_lists, active_sources, intent
        )
        p[d] = calibrated_relevance_probability(features, params)

    # 5. Initialize output with legacy spine
    output = list(legacy[:20])

    # 6. Optional prefix lock
    k_lock = params.prefix_lock[intent]
    locked = set(output[:k_lock])

    # 7. Candidate insertion
    candidates = sorted(pool, key=lambda d: p[d], reverse=True)

    for d in candidates:
        if d in output:
            # allow promotion only if it passes pairwise/DCG test
            output = maybe_promote_existing(d, output, p, intent, params, locked)
            continue

        if not admission_precheck(d, state, intent, p, source_lists, params):
            continue

        best_move = None
        best_gain = 0.0

        for j in allowed_positions(intent, params):
            if j < k_lock:
                continue

            gain = expected_dcg_delta_with_penalty(
                d=d,
                j=j,
                output=output,
                p=p,
                legacy=legacy,
                intent=intent,
                params=params,
            )

            if gain > best_gain:
                best_gain = gain
                best_move = j

        if best_move is not None and best_gain > params.insert_threshold[intent][best_move]:
            if quota_ok(d, best_move, output, legacy, intent, params):
                if demotion_budget_ok(d, best_move, output, legacy, p, intent, params):
                    output = insert_at(output, d, best_move, max_len=20)

    return dedupe_keep_order(output)[:20]
